При $x \to 0$
\begin{equation}
    \begin{aligned}
        &sin(x) \approx x \\
        &tg(x) \approx x \\
        &ln(1+x) \approx x \\
        &a^x - 1 \approx xln(a) \\
        &e^x - 1 \approx x \\
        &(1 + x)^\alpha - 1 \approx \alpha x \\
        &arcsin(x) \approx x \\
        &arctg(x) \approx x \\
    \end{aligned}
\end{equation}
